    const main_window_id = c.SDL_GetWindowID(window);
    var running = true;
    var win_w: f32 = 800;
    var win_h: f32 = 600;

    while (running) {
        var event: c.SDL_Event = undefined;
        while (c.SDL_PollEvent(&event) != 0) {
            const evt_window_id: u32 = switch (event.type) {
                c.SDL_WINDOWEVENT => event.window.windowID,
                c.SDL_MOUSEMOTION => event.motion.windowID,
                c.SDL_MOUSEBUTTONDOWN, c.SDL_MOUSEBUTTONUP => event.button.windowID,
                c.SDL_MOUSEWHEEL => event.wheel.windowID,
                c.SDL_KEYDOWN, c.SDL_KEYUP => event.key.windowID,
                else => main_window_id,
            };
            if (evt_window_id != main_window_id) {
                if (win_mgr.findByWindowId(evt_window_id)) |win_idx| {
                    switch (event.type) {
                        c.SDL_WINDOWEVENT => {
                            if (event.window.event == c.SDL_WINDOWEVENT_CLOSE) { win_mgr.handleClose(win_idx); }
                            else if (event.window.event == c.SDL_WINDOWEVENT_SIZE_CHANGED) { win_mgr.handleResize(win_idx, event.window.data1, event.window.data2); }
                        },
                        c.SDL_MOUSEMOTION => { win_mgr.handleMouseMotion(win_idx, @floatFromInt(event.motion.x), @floatFromInt(event.motion.y)); },
                        c.SDL_MOUSEBUTTONDOWN => { win_mgr.handleClick(win_idx, @floatFromInt(event.button.x), @floatFromInt(event.button.y)); },
                        c.SDL_MOUSEWHEEL => {
                            var mx_i: c_int = undefined; var my_i: c_int = undefined;
                            _ = c.SDL_GetMouseState(&mx_i, &my_i);
                            win_mgr.handleWheel(win_idx, @floatFromInt(mx_i), @floatFromInt(my_i), @floatFromInt(event.wheel.y));
                        },
                        else => {},
                    }
                    continue;
                }
            }
            switch (event.type) {
                c.SDL_QUIT => running = false,
                c.SDL_WINDOWEVENT => { if (event.window.event == c.SDL_WINDOWEVENT_SIZE_CHANGED) { win_w = @floatFromInt(event.window.data1); win_h = @floatFromInt(event.window.data2); } },
                c.SDL_TEXTINPUT => {
                    input_mod.handleTextInput(@ptrCast(&event.text.text));
                },
                c.SDL_KEYDOWN => {
                    const ctrl = (event.key.keysym.mod & c.KMOD_CTRL) != 0;
                    // Route to focused input first
                    if (input_mod.getFocusedId() != null) {
                        if (ctrl and input_mod.handleCtrlKey(event.key.keysym.sym)) {
                            // Ctrl+A/C/X/V/Z handled
                        } else if (event.key.keysym.sym == c.SDLK_ESCAPE) { input_mod.unfocus(); }
                        else { _ = input_mod.handleKey(event.key.keysym.sym); }
                    } else if (ctrl and event.key.keysym.sym == c.SDLK_a) {
                        // Ctrl+A — select all text in current node, or find first text node
                        if (sel_node) |sn| {
                            if (sn.text) |txt| {
                                sel_start = 0;
                                sel_end = txt.len;
                            }
                        } else if (hovered_node) |hn| {
                            if (hn.text) |txt| {
                                sel_node = hn;
                                sel_start = 0;
                                sel_end = txt.len;
                            }
                        }
                    } else if (ctrl and event.key.keysym.sym == c.SDLK_c) {
                        // Ctrl+C — copy selected text to clipboard
                        if (sel_node) |sn| {
                            if (sn.text) |txt| {
                                const s0 = @min(sel_start, sel_end);
                                const s1 = @max(sel_start, sel_end);
                                if (s1 > s0 and s1 <= txt.len) {
                                    // SDL_SetClipboardText needs null-terminated string
                                    var clip_buf: [4096]u8 = undefined;
                                    const clip_len = @min(s1 - s0, clip_buf.len - 1);
                                    @memcpy(clip_buf[0..clip_len], txt[s0..s0 + clip_len]);
                                    clip_buf[clip_len] = 0;
                                    _ = c.SDL_SetClipboardText(@ptrCast(&clip_buf));
                                }
                            }
                        }
                    } else if (event.key.keysym.sym == c.SDLK_ESCAPE) { running = false; }
                    else { if (hovered_node) |node| { if (node.handlers.on_key) |handler| handler(event.key.keysym.sym); } }
                },
                c.SDL_MOUSEMOTION => {
                    const mx: f32 = @floatFromInt(event.motion.x); const my: f32 = @floatFromInt(event.motion.y);
                    const prev = hovered_node; hovered_node = events.hitTest(&root, mx, my);
                    if (prev != hovered_node) { if (prev) |p| { if (p.handlers.on_hover_exit) |h| h(); } if (hovered_node) |n| { if (n.handlers.on_hover_enter) |h| h(); } }
                    // Text selection drag
                    if (sel_dragging) {
                        if (sel_node) |sn| {
                            if (sn.text) |txt| {
                                const pad_l = sn.style.padLeft();
                                const pad_r = sn.style.padRight();
                                const pad_t = sn.style.padTop();
                                const local_x = mx - sn.computed.x - pad_l;
                                const local_y = my - sn.computed.y - pad_t;
                                const text_max_w = sn.computed.w - pad_l - pad_r;
                                sel_end = g_text_engine.?.hitTestWrapped(txt, local_x, local_y, sn.font_size, text_max_w);
                            }
                        }
                    }
                },
                c.SDL_MOUSEBUTTONDOWN => {
                    const mx: f32 = @floatFromInt(event.button.x); const my: f32 = @floatFromInt(event.button.y);
                    if (events.hitTest(&root, mx, my)) |node| {
                        // Focus TextInput if clicked
                        if (node.input_id) |iid| {
                            input_mod.focus(iid);
                            sel_node = null;
                        } else {
                            // Clicked non-input — unfocus and fire press handler
                            input_mod.unfocus();
                            if (node.handlers.on_press) |handler| handler();
                            // Text selection — start on text nodes
                            if (node.text) |txt| {
                                const pad_l = node.style.padLeft();
                                const pad_r = node.style.padRight();
                                const pad_t = node.style.padTop();
                                const local_x = mx - node.computed.x - pad_l;
                                const local_y = my - node.computed.y - pad_t;
                                const text_max_w = node.computed.w - pad_l - pad_r;
                                const now = c.SDL_GetTicks();
                                const byte_idx = g_text_engine.?.hitTestWrapped(txt, local_x, local_y, node.font_size, text_max_w);
                                // Multi-click detection
                                if (sel_node == node and now - sel_last_click < 400) {
                                    sel_click_count += 1;
                                } else {
                                    sel_click_count = 1;
                                }
                                sel_last_click = now;
                                if (sel_click_count >= 3) {
                                    // Triple click — select all text in node
                                    sel_node = node;
                                    sel_start = 0;
                                    sel_end = txt.len;
                                    sel_anchor = 0;
                                    sel_dragging = false;
                                } else if (sel_click_count == 2) {
                                    // Double click — select word
                                    const wb = TextEngine.wordBoundsAt(txt, byte_idx);
                                    sel_node = node;
                                    sel_start = wb.start;
                                    sel_end = wb.end;
                                    sel_anchor = wb.start;
                                    sel_dragging = false;
                                } else {
                                    // Single click — start cursor selection
                                    sel_node = node;
                                    sel_anchor = byte_idx;
                                    sel_start = byte_idx;
                                    sel_end = byte_idx;
                                    sel_dragging = true;
                                }
                            } else {
                                sel_node = null;
                                sel_start = 0;
                                sel_end = 0;
                            }
                        }
                    } else {
                        input_mod.unfocus();
                        sel_node = null;
                        sel_start = 0;
                        sel_end = 0;
                    }
                },
                c.SDL_MOUSEBUTTONUP => {
                    sel_dragging = false;
                },
                c.SDL_MOUSEWHEEL => {
                    var mx_i: c_int = undefined; var my_i: c_int = undefined;
                    _ = c.SDL_GetMouseState(&mx_i, &my_i);
                    const mx: f32 = @floatFromInt(mx_i); const my: f32 = @floatFromInt(my_i);
                    if (events.findScrollContainer(&root, mx, my)) |scroll_node| {
                        scroll_node.scroll_y -= @as(f32, @floatFromInt(event.wheel.y)) * 30.0;
                        const max_scroll = @max(0.0, scroll_node.content_height - scroll_node.computed.h);
                        scroll_node.scroll_y = @max(0.0, @min(scroll_node.scroll_y, max_scroll));
                    }
                },
                else => {},
            }
        }
